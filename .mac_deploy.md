# macOS Deploy Notes

## 1. Extract The Source Tarball

Run these commands from the directory where the downloaded tarball lives:

```bash
cd ~/fjarcode-core
rm -rf fjarcode-27.0.0
tar -xzf fjarcode-core-source-v27.0.0-20260309.tar.gz
cd fjarcode-27.0.0
```

You should now see files like:

```bash
configure
build-aux
src
Makefile.am
```

## 2. Install Build Dependencies

```bash
brew install autoconf automake libtool pkgconf make qt@5 boost libevent zeromq sqlite qrencode
```

## 3. Configure And Build

Do not run `autogen.sh` on the published source tarball.

```bash
export PATH="$(brew --prefix make)/libexec/gnubin:$(brew --prefix qt@5)/bin:$PATH"
export QTDIR="$(brew --prefix qt@5)"
export MAKE=gmake
export PKG_CONFIG_PATH="$(brew --prefix qt@5)/lib/pkgconfig:$(brew --prefix libevent)/lib/pkgconfig:$(brew --prefix zeromq)/lib/pkgconfig:$(brew --prefix sqlite)/lib/pkgconfig:$(brew --prefix qrencode)/lib/pkgconfig"

./configure MAKE=gmake \
  --with-gui=qt5 \
  --without-miniupnpc \
  --disable-tests \
  --disable-bench \
  --disable-dependency-tracking \
  CFLAGS="-O2" \
  CXXFLAGS="-O2"

gmake -j"$(sysctl -n hw.ncpu)"
```

## 4. Create The macOS App Bundle

This bundle script uses `fjarcode.png` as the icon source and converts it into a proper `.icns` file for macOS.

```bash
set -e

export PATH="$(brew --prefix qt@5)/bin:$PATH"

APP="FJARCODE-Qt.app"
ICON_PNG="fjarcode.png"
ICONSET="FJARCODE.iconset"

rm -rf "$APP" dist "$ICONSET"

mkdir -p "$APP/Contents/MacOS" "$APP/Contents/Resources"
mkdir -p "$ICONSET"

sips -z 16 16     "$ICON_PNG" --out "$ICONSET/icon_16x16.png"
sips -z 32 32     "$ICON_PNG" --out "$ICONSET/icon_16x16@2x.png"
sips -z 32 32     "$ICON_PNG" --out "$ICONSET/icon_32x32.png"
sips -z 64 64     "$ICON_PNG" --out "$ICONSET/icon_32x32@2x.png"
sips -z 128 128   "$ICON_PNG" --out "$ICONSET/icon_128x128.png"
sips -z 256 256   "$ICON_PNG" --out "$ICONSET/icon_128x128@2x.png"
sips -z 256 256   "$ICON_PNG" --out "$ICONSET/icon_256x256.png"
sips -z 512 512   "$ICON_PNG" --out "$ICONSET/icon_256x256@2x.png"
sips -z 512 512   "$ICON_PNG" --out "$ICONSET/icon_512x512.png"
sips -z 1024 1024 "$ICON_PNG" --out "$ICONSET/icon_512x512@2x.png"

iconutil -c icns "$ICONSET" -o "$APP/Contents/Resources/FJARCODE.icns"

cp src/qt/fjarcode-qt "$APP/Contents/MacOS/FJARCODE-Qt"

cat > "$APP/Contents/Info.plist" <<'PLIST'
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>CFBundleDevelopmentRegion</key>
  <string>English</string>
  <key>CFBundleExecutable</key>
  <string>FJARCODE-Qt</string>
  <key>CFBundleIdentifier</key>
  <string>org.fjarcode.qt</string>
  <key>CFBundleIconFile</key>
  <string>FJARCODE.icns</string>
  <key>CFBundleInfoDictionaryVersion</key>
  <string>6.0</string>
  <key>CFBundleName</key>
  <string>FJARCODE-Qt</string>
  <key>CFBundleDisplayName</key>
  <string>FJARCODE-Qt</string>
  <key>CFBundlePackageType</key>
  <string>APPL</string>
  <key>CFBundleShortVersionString</key>
  <string>27.0.0</string>
  <key>CFBundleVersion</key>
  <string>27.0.0</string>
  <key>LSMinimumSystemVersion</key>
  <string>12.0</string>
  <key>NSHighResolutionCapable</key>
  <true/>
</dict>
</plist>
PLIST

python3 contrib/macdeploy/macdeployqtplus \
  "$APP" \
  FJARCODE-Qt \
  -translations-dir "$(brew --prefix qt@5)/translations"

touch "$APP"
touch "$APP/Contents/Info.plist"

codesign --force --deep --sign - dist/FJARCODE-Qt.app

rm -rf "$ICONSET"

echo "DONE"
echo "Bundle created in dist/"
```

## 5. Quick Checks

If `./configure` says it cannot be found, you are not inside the extracted `fjarcode-27.0.0` directory yet.

If `configure` reports missing auxiliary files, the source tarball is wrong and should be re-downloaded.

If `macdeployqtplus` errors on translations, keep using:

```bash
-translations-dir "$(brew --prefix qt@5)/translations"
```
